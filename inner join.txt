-- create a table
create table student(student_id int(20) primary key,student_name varchar(50),
student_class varchar(50));

-- fetch some values
create table studentmarks(student_id int not null,student_sub1 int(20),student_sub2 varchar(50),
foreign key(student_id)references student(student_id));
